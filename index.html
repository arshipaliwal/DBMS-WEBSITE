Exception Handling

DECLARE
    -- Variables
    v_empno EMP.EMPNO%TYPE := 9999; -- Use a non-existing EMPNO for NO_DATA_FOUND
    v_sal EMP.SAL%TYPE := 1500;     -- Example salary for validation
    v_age NUMBER := 16;              -- Example age for validation
    v_deptno NUMBER := 99;           -- Invalid department number
    v_result NUMBER;
    v_err_msg VARCHAR2(200);
    
    -- User-defined exceptions
    e_salary_low EXCEPTION;
    e_age_invalid EXCEPTION;
    e_dept_invalid EXCEPTION;
    e_salary_high EXCEPTION;

BEGIN
    -- 1. NO_DATA_FOUND
    BEGIN
        SELECT SAL INTO v_result FROM EMP WHERE EMPNO = v_empno;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            DBMS_OUTPUT.PUT_LINE('No employee found with EMPNO ' || v_empno);
    END;

    -- 2. TOO_MANY_ROWS
    BEGIN
        SELECT SAL INTO v_result FROM EMP; -- Assume multiple rows exist
    EXCEPTION
        WHEN TOO_MANY_ROWS THEN
            DBMS_OUTPUT.PUT_LINE('Error: Too many rows returned');
    END;

    -- 3. ZERO_DIVIDE
    BEGIN
        v_result := 10 / 0;
    EXCEPTION
        WHEN ZERO_DIVIDE THEN
            DBMS_OUTPUT.PUT_LINE('Error: Division by zero');
    END;

    -- 4. VALUE_ERROR
    BEGIN
        v_result := TO_NUMBER('ABC');
    EXCEPTION
        WHEN VALUE_ERROR THEN
            DBMS_OUTPUT.PUT_LINE('Error: Value conversion issue');
    END;

    -- 5. WHEN OTHERS
    BEGIN
        v_result := 10 / NULL;
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Generic error occurred: ' || SQLERRM);
    END;

    -- 6 & 7. SQLCODE & SQLERRM
    BEGIN
        v_result := 10 / 0;
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Error Code: ' || SQLCODE || ', Message: ' || SQLERRM);
    END;

    -- 8. Logging exception
    BEGIN
        v_result := 10 / 0;
    EXCEPTION
        WHEN OTHERS THEN
            v_err_msg := SQLERRM;
            DBMS_OUTPUT.PUT_LINE('Logged error: ' || v_err_msg);
    END;

    -- 9. RAISE_APPLICATION_ERROR for salary
    IF v_sal < 2000 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Salary cannot be less than 2000');
    END IF;

    -- 10. Validate age
    IF v_age < 18 THEN
        RAISE_APPLICATION_ERROR(-20002, 'Employee age must be at least 18');
    END IF;

    -- 11. Restrict update
    IF v_sal < 0 THEN
        RAISE_APPLICATION_ERROR(-20003, 'Cannot update salary to negative value');
    END IF;

    -- 12. Invalid dept number
    IF v_deptno NOT IN (10,20,30) THEN
        RAISE_APPLICATION_ERROR(-20004, 'Invalid department number');
    END IF;

    -- 13. User-defined exception
    IF v_sal < 1000 THEN
        RAISE e_salary_low;
    END IF;

    -- 14. Salary exceeds limit
    IF v_sal > 10000 THEN
        RAISE e_salary_high;
    END IF;

    -- 15. Input validation
    IF v_age < 18 THEN
        RAISE e_age_invalid;
    END IF;

    -- 16. Multiple user-defined exceptions
    BEGIN
        IF v_sal < 1000 THEN
            RAISE e_salary_low;
        ELSIF v_age < 18 THEN
            RAISE e_age_invalid;
        END IF;
    EXCEPTION
        WHEN e_salary_low THEN
            DBMS_OUTPUT.PUT_LINE('Salary too low');
        WHEN e_age_invalid THEN
            DBMS_OUTPUT.PUT_LINE('Age not valid');
    END;

    -- 17 & 18. Re-raise exception
    BEGIN
        BEGIN
            v_result := 10 / 0;
        EXCEPTION
            WHEN ZERO_DIVIDE THEN
                DBMS_OUTPUT.PUT_LINE('Inner block handled division by zero');
                RAISE; -- Re-raise to outer block
        END;
    EXCEPTION
        WHEN ZERO_DIVIDE THEN
            DBMS_OUTPUT.PUT_LINE('Outer block caught re-raised exception');
    END;

    -- 19. Log and re-raise
    BEGIN
        BEGIN
            v_result := 10 / 0;
        EXCEPTION
            WHEN ZERO_DIVIDE THEN
                DBMS_OUTPUT.PUT_LINE('Logging division by zero error');
                RAISE;
        END;
    EXCEPTION
        WHEN ZERO_DIVIDE THEN
            DBMS_OUTPUT.PUT_LINE('Caught in outer block after logging');
    END;

    -- 20. Combined exception handling while fetching employee
    BEGIN
        SELECT SAL INTO v_result FROM EMP WHERE EMPNO = v_empno;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            DBMS_OUTPUT.PUT_LINE('Employee not found');
        WHEN TOO_MANY_ROWS THEN
            DBMS_OUTPUT.PUT_LINE('Multiple employees found with same EMPNO');
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Other error: ' || SQLERRM);
    END;

    -- 21. User-defined + RAISE_APPLICATION_ERROR
    IF v_sal < 2000 THEN
        RAISE_APPLICATION_ERROR(-20005, 'Custom Error: Salary too low');
    END IF;

    -- 22. Validate salary
    IF v_sal < 0 THEN
        RAISE_APPLICATION_ERROR(-20006, 'Salary cannot be negative');
    END IF;

    -- 23. Division with SQLCODE & SQLERRM
    BEGIN
        v_result := 10 / 0;
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM || ', Code: ' || SQLCODE);
    END;

    -- 24. Simulate bank withdrawal
    DECLARE
        v_balance NUMBER := 500;
        v_withdraw NUMBER := 600;
    BEGIN
        IF v_withdraw > v_balance THEN
            RAISE_APPLICATION_ERROR(-20007, 'Insufficient balance');
        ELSE
            v_balance := v_balance - v_withdraw;
            DBMS_OUTPUT.PUT_LINE('Withdrawal successful. New balance: ' || v_balance);
        END IF;
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Bank Error: ' || SQLERRM);
    END;

    -- 25. Multiple validations
    BEGIN
        IF v_sal < 1000 THEN
            RAISE e_salary_low;
        ELSIF v_age < 18 THEN
            RAISE e_age_invalid;
        ELSE
            DBMS_OUTPUT.PUT_LINE('All validations passed');
        END IF;
    EXCEPTION
        WHEN e_salary_low THEN
            DBMS_OUTPUT.PUT_LINE('Salary too low');
        WHEN e_age_invalid THEN
            DBMS_OUTPUT.PUT_LINE('Age not valid');
    END;

    -- 26. Log errors into variable
    BEGIN
        v_result := 10 / 0;
    EXCEPTION
        WHEN OTHERS THEN
            v_err_msg := SQLERRM;
            DBMS_OUTPUT.PUT_LINE('Error logged: ' || v_err_msg);
    END;

    -- 27. Nested exception blocks with re-raise
    BEGIN
        BEGIN
            v_result := 10 / 0;
        EXCEPTION
            WHEN ZERO_DIVIDE THEN
                DBMS_OUTPUT.PUT_LINE('Inner block caught division by zero');
                RAISE;
        END;
    EXCEPTION
        WHEN ZERO_DIVIDE THEN
            DBMS_OUTPUT.PUT_LINE('Outer block re-raised caught');
    END;

END;
